# ETL1
files = glob.glob('/content/drive/My Drive//DL_data/ETL1/*')
for fname in files:
    if fname == "/content/drive/My Drive//DL_data/ETL1/ETL1INFO":
        continue
    print(fname)
    # ETL1
    f = open(fname, 'rb')
    f.seek(0)
    while True:
        # 
        s = f.read(2052)
        if not s:
            break
        # Python
        r = struct.unpack('>H2sH6BI4H4B4x2016s4x', s)
        code_ascii = r[1]
        code_jis = r[3]
        # 
        iF = Image.frombytes('F', (64, 63), r[18], 'bit', 4)
        iP = iF.convert('L')
        # 
        dir = outdir + '/' + str(code_jis)
        if not os.path.exists(dir):
            os.mkdir(dir)
        fn = "{0:02x}-{1:02x}-{2:04x}.png".format(code_jis, r[0], r[2])
        fullpath = dir + '/' + fn
        if os.path.exists(fullpath):
            continue
        enhancer = ImageEnhance.Brightness(iP)
        iE = enhancer.enhance(16)
        iE.save(fullpath, 'PNG')
